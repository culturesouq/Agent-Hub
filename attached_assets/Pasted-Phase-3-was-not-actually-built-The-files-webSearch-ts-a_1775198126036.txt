export interface SearchResult {
  title: string;
  snippet: string;
  url: string;
}

export async function webSearch(query: string): Promise<SearchResult[]> {
  const apiKey = process.env.SERPER_API_KEY;
  if (!apiKey) return [];

  try {
    const res = await fetch('https://google.serper.dev/search', {
      method: 'POST',
      headers: { 'X-API-KEY': apiKey, 'Content-Type': 'application/json' },
      body: JSON.stringify({ q: query, num: 3 }),
    });
    const data = await res.json() as any;
    return ([data.org](https://data.org)anic ?? [])
      .filter((r: any) => r.snippet)
      .slice(0, 3)
      .map((r: any) => ({ title: r.title ?? '', snippet: r.snippet, url: r.link ?? '' }));
  } catch {
    return [];
  }
}

import { chatCompletion } from './openrouter.js';
import { embed } from './vectorSearch.js';
import { db } from '@workspace/db';
import { operatorKbTable } from '@workspace/db';
import { and, eq, sql } from 'drizzle-orm';

const DISTILL_MODEL = 'meta-llama/llama-3.3-70b-instruct';

export async function verifyAndStore(
  operatorId: string,
  ownerId: string,
  content: string,
  sourceUrl: string | undefined,
  sourceName: string,
  mandate: string,
): Promise<{ stored: boolean; status: string; reason?: string }> {
  try {
    // 1. Relevance
    const relText = await chatCompletion([
      { role: 'user', content: `Rate relevance 0-100. Mandate: "${mandate}". Content: "${content}". Reply with a number only.` }
    ], DISTILL_MODEL);
    const relevance = parseInt(relText.trim(), 10);
    if (isNaN(relevance) || relevance < 50) return { stored: false, status: 'skipped', reason: 'low relevance' };

    // 2. Durability
    const durText = await chatCompletion([
      { role: 'user', content: `Is this fact stable over time or ephemeral (price, today's news, live event)? Reply "stable" or "ephemeral" only.\nContent: "${content}"` }
    ], DISTILL_MODEL);
    if (durText.trim().toLowerCase().includes('ephemeral')) return { stored: false, status: 'skipped', reason: 'ephemeral' };

    // 3. Privacy
    const privText = await chatCompletion([
      { role: 'user', content: `Does this contain personal identifiable information (names, emails, phone numbers, addresses)? Reply "yes" or "no" only.\nContent: "${content}"` }
    ], DISTILL_MODEL);
    if (privText.trim().toLowerCase().startsWith('yes')) return { stored: false, status: 'skipped', reason: 'PII detected' };

    // 4. Deduplication + 5. Contradiction
    const embedding = await embed(content);
    const similar = await db.execute(sql`
      SELECT content, verification_status,
             (embedding <=> ${JSON.stringify(embedding)}::vector) as distance
      FROM operator_kb
      WHERE operator_id = ${operatorId}
      ORDER BY distance ASC LIMIT 3
    `);

    for (const row of similar.rows as any[]) {
      if (row.distance < 0.08) return { stored: false, status: 'skipped', reason: 'duplicate' };
    }

    let verificationStatus = relevance >= 85 ? 'verified' : 'pending';
    let flagReason: string | undefined;
    for (const row of similar.rows as any[]) {
      if (row.distance < 0.15 && row.verification_status === 'verified') {
        verificationStatus = 'pending';
        flagReason = 'Possible contradiction with existing knowledge';
        break;
      }
    }

    // 6. Confidence (= relevance) already checked above

    // 7. Source trust
    const trustedDomains = ['.gov', '.edu', '[reuters.com](https://reuters.com)', '[bbc.com](https://bbc.com)', '[nature.com](https://nature.com)', 'who.int', '[un.org](https://un.org)', '[bloomberg.com](https://bloomberg.com)'];
    const isTrusted = sourceUrl && trustedDomains.some(d => sourceUrl.includes(d));
    const sourceTrustLevel = isTrusted ? 'external_verified' : 'external_unverified';

    // Store
    await db.execute(sql`
      INSERT INTO operator_kb
        (id, operator_id, owner_id, content, embedding, source_name, source_url,
         source_trust_level, confidence_score, intake_tags, is_pipeline_intake,
         privacy_cleared, content_cleared, verification_status, chunk_index, created_at)
      VALUES
        (${crypto.randomUUID()}, ${operatorId}, ${ownerId}, ${content},
         ${JSON.stringify(embedding)}::vector, ${sourceName}, ${sourceUrl ?? null},
         ${sourceTrustLevel}, ${relevance}, '["web_search"]', true, true, true,
         ${verificationStatus}, 0, NOW())
    `);

    if (flagReason) {
      await db.execute(sql`
        UPDATE operator_kb SET flag_reason = ${flagReason}
        WHERE operator_id = ${operatorId} AND content = ${content}
      `);
    }

    return { stored: true, status: verificationStatus };
  } catch (e) {
    return { stored: false, status: 'skipped', reason: (e as Error).message };
  }
}
